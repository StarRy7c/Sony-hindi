    // Global variables
    let chatData = { users: [] };
    let onlineUsers = 1247;

    let channelLinks = [];

    // Parse URL parameters to get channel links
    function parseChannelLinks() {
      const urlParams = new URLSearchParams(window.location.search);
      const links = [];
      
      // Look for link1, link2, link3, etc. parameters
      let linkIndex = 1;
      while (urlParams.has(`link${linkIndex}`)) {
        const link = urlParams.get(`link${linkIndex}`);
        if (link) {
          links.push({
            url: decodeURIComponent(link),
            name: `Channel ${linkIndex}`
          });
        }
        linkIndex++;
      }
      
      // Also check for direct link parameters (link1, link2, etc.)
      urlParams.forEach((value, key) => {
        if (key.match(/^link\d+$/)) {
          const index = parseInt(key.replace('link', ''));
          if (!links[index - 1]) { // Only add if not already added
            links.push({
              url: decodeURIComponent(value),
              name: `Channel ${index}`
            });
          }
        }
      });

      return links;
